import { createWorker } from 'tesseract.js'; let cvPromise; let workerPromise; function loadOpenCv() { if (window.cv?.Mat) return Promise.resolve(window.cv); if (cvPromise) return cvPromise; cvPromise = new Promise((resolve, reject) => { const script = document.createElement('script'); script.src = 'https://docs.opencv.org/4.x/opencv.js'; script.async = true; script.onload = async () => { const started = Date.now(); while (Date.now() - started < 20000) { const cv = window.cv && typeof window.cv.then === 'function' ? await window.cv : window.cv; if (cv?.Mat) return resolve(cv); await new Promise(r => setTimeout(r, 120)); } reject(new Error('OpenCV neielādējās.')); }; script.onerror = () => reject(new Error('OpenCV skripts neielādējās.')); document.head.appendChild(script); }); return cvPromise; } async function getWorker(onProgress) { if (!workerPromise) { workerPromise = createWorker('eng', 1, { logger: message => { if (message.status === 'recognizing text') onProgress?.(Math.round((message.progress || 0) * 100)); } }).then(async worker => { await worker.setParameters({ tessedit_char_whitelist: 'ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789', tessedit_pageseg_mode: '7' }); return worker; }); } return workerPromise; } function score(value, confidence) { if (value.length < 4 || value.length > 9) return -100; let result = confidence / 10 - Math.abs(6 - value.length); if (/[A-Z]/.test(value) && /\d/.test(value)) result += 10; if (/^[A-Z]{1,3}\d{1,4}$/.test(value)) result += 12; if (/^[A-Z]{4,9}$/.test(value)) result -= 20; return result; } function candidates(text) { const values = String(text || '').toUpperCase().split(/\s+/) .map(v => v.replace(/[^A-Z0-9]/g, '')) .filter(v => v.length >= 4 && v.length <= 9); const joined = String(text || '').toUpperCase().replace(/[^A-Z0-9]/g, ''); if (joined.length >= 4 && joined.length <= 9) values.push(joined); return [...new Set(values)]; } function findPlate(cv, canvas, previewCanvas) { const src = cv.imread(canvas), gray = new cv.Mat(), blur = new cv.Mat(), edges = new cv.Mat(); const closed = new cv.Mat(), contours = new cv.MatVector(), hierarchy = new cv.Mat(); const kernel = cv.getStructuringElement(cv.MORPH_RECT, new cv.Size(5, 3)); let best = null, bestScore = -Infinity; try { cv.cvtColor(src, gray, cv.COLOR_RGBA2GRAY); cv.bilateralFilter(gray, blur, 9, 75, 75, cv.BORDER_DEFAULT); cv.Canny(blur, edges, 60, 180); cv.morphologyEx(edges, closed, cv.MORPH_CLOSE, kernel, new cv.Point(-1, -1), 2); cv.findContours(closed, contours, hierarchy, cv.RETR_LIST, cv.CHAIN_APPROX_SIMPLE); const imageArea = src.cols * src.rows; for (let i = 0; i < contours.size(); i++) { const contour = contours.get(i); const area = Math.abs(cv.contourArea(contour)); const rect = cv.boundingRect(contour); contour.delete(); const aspect = rect.width / Math.max(1, rect.height); if (area < imageArea * .0015 || area > imageArea * .25 || rect.width < 80 || rect.height < 20 || aspect < 2 || aspect > 7) continue; const cx = rect.x + rect.width / 2, cy = rect.y + rect.height / 2; const center = 1 - Math.min(.8, Math.abs(cx - src.cols/2)/src.cols + Math.abs(cy - src.rows/2)/src.rows); const current = 6/(1 + Math.abs(aspect - 4.5)) + center * 4 + Math.min(2, area/(imageArea*.02)); if (current > bestScore) { bestScore = current; best = rect; } } if (!best) throw new Error('Numurzīmes taisnstūris nav atrasts.'); const px = Math.round(best.width*.08), py = Math.round(best.height*.25); const x = Math.max(0, best.x-px), y = Math.max(0, best.y-py); const w = Math.min(src.cols-x, best.width+px*2), h = Math.min(src.rows-y, best.height+py*2); const roi = src.roi(new cv.Rect(x,y,w,h)); const enlarged = new cv.Mat(), g = new cv.Mat(), normalized = new cv.Mat(), threshold = new cv.Mat(); cv.resize(roi, enlarged, new cv.Size(1100, Math.round(1100*h/w)), 0, 0, cv.INTER_CUBIC); cv.cvtColor(enlarged, g, cv.COLOR_RGBA2GRAY); cv.equalizeHist(g, normalized); cv.adaptiveThreshold(normalized, threshold, 255, cv.ADAPTIVE_THRESH_GAUSSIAN_C, cv.THRESH_BINARY, 31, 11); cv.imshow(previewCanvas, threshold); const colorCanvas = document.createElement('canvas'); cv.imshow(colorCanvas, enlarged); const binaryCanvas = document.createElement('canvas'); cv.imshow(binaryCanvas, threshold); roi.delete(); enlarged.delete(); g.delete(); normalized.delete(); threshold.delete(); return [binaryCanvas, colorCanvas]; } finally { src.delete(); gray.delete(); blur.delete(); edges.delete(); closed.delete(); contours.delete(); hierarchy.delete(); kernel.delete(); } } export async function recognizePlate(frameCanvas, previewCanvas, onProgress) { const cv = await loadOpenCv(); const images = findPlate(cv, frameCanvas, previewCanvas); const worker = await getWorker(onProgress); let best = { plate: '', score: -Infinity, confidence: 0 }; for (const image of images) { const result = await worker.recognize(image); for (const candidate of candidates(result.data.text)) { const candidateScore = score(candidate, Number(result.data.confidence || 0)); if (candidateScore > best.score) best = { plate: candidate, score: candidateScore, confidence: result.data.confidence }; } } if (!/[A-Z]/.test(best.plate) || !/\d/.test(best.plate)) return { plate: '', confidence: best.confidence }; return best; } export const anprProviders = { browser: { id: 'browser', label: 'Bezmaksas OCR ierīcē', recognize: recognizePlate }, external: { id: 'external', label: 'Ārējs ANPR (nākotnei)', recognize: async () => { throw new Error('Ārējais ANPR vēl nav konfigurēts.'); } } };
